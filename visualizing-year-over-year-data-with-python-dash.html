from faker import Faker
import pandas as pd
from itertools import product
import datetime


def schoolyear4digit(date):
    year = date.year
    if date.month >= 7:
        year += 1
    return year


def random_value(metric):
    options = {
        "left_digits": 3,
        "right_digits": 2,
        "positive": True,
        "min_value": 0,
        "max_value": 100,
    }
    if metric == "Chronic Absence":
        options["max_value"] = 25
    elif metric == "ADA":
        options["min_value"] = 89
    elif metric == "Enrollment":
        options["min_value"] = 92
        options["max_value"] = 105
    return f.pyfloat(**options)

f = Faker()

today = datetime.date.today()
metrics = ["ADA", "Chronic Absence", "Enrollment"]
schools = ["Justice HS", "Freedom HS"]
months = list(pd.date_range(start="2019-08-01", end=today, freq="M"))
school_metrics = list(product(metrics, schools, months))

data = [
    {
        "metric": metric,
        "school": school,
        "date": date,
        "month": date.month,
        "month_numeric": date.month - 6 if date.month >=7 else date.month + 6,
        "value": random_value(metric),
        "school_year": schoolyear4digit(date),
    } for metric, school, date in school_metrics
]
df = pd.DataFrame(data)
print(df)
